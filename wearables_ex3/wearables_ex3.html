1. Segmenting¶

We know that the signals' sampling frequency is 200Hz, so utilizing this information, we can divide the data into 10s segments, because 200 samples/s = 2000 samples/10s

In [6]:
def segmentor(data, fr, segment_length):
    x = 0
    segments = []
    while x<len(data):
        segment = data.loc[x:(x+fr*segment_length-1)].reset_index(drop=True)
        segments.append(segment)
        x = x + len(segment)
    return segments

def average_waveform(segments, fr, time):
    peak_is = []
    peaks = []
    z = 0
    average_peak_locations = []
    for i in segments:
        peak_is.append(sp.signal.find_peaks(i, height=1.00))
        peaks_at_z =[]
        peak_locs = []
        for j in peak_is[z][0]:
            peak_locs.append(j)
            peaks_at_z.append(segments[z][j])
        peaks.append(peaks_at_z)
        average_peak_locations.append(peak_locs)
        z+=1
    peaks.remove(peaks[len(peaks)-1])
    
    df1 = pd.DataFrame(peaks)
    df2 = pd.DataFrame(average_peak_locations)
    indices_mean = df2.mean(axis=0).astype(int)
    peaks_mean = df1.mean(axis=0)
    
    average_waveform = []
    x = 0
    for i in range(0, fr*time):
        if x<len(indices_mean):
            if i == indices_mean.loc[x]:
                average_waveform.append(peaks_mean[x])
                x+=1
            else:
                average_waveform.append(0)
        else:
            average_waveform.append(0)
        
    return average_waveform
